259 inline void clear_deleteme_list(const bool murder=false)
260 {
261 if (!this->deleteme.empty()) {
262 // It's possible we could add items to this list while
263 // running Python code if there's a thread switch, so we
264 // need to defensively copy it before that can happen.
265 deleteme_t copy = this->deleteme;
266 this->deleteme.clear(); // in case things come back on the list
267 for(deleteme_t::iterator it = copy.begin(), end = copy.end();
268 it != end;
269 ++it ) {
270 PyGreenlet* to_del = *it;
271 if (murder) {
272 // Force each greenlet to appear dead; we can't raise an
273 // exception into it anymore anyway.
274 to_del->pimpl->murder_in_place();
275 }
276
277 // The only reference to these greenlets should be in
278 // this list, decreffing them should let them be
279 // deleted again, triggering calls to green_dealloc()
280 // in the correct thread (if we're not murdering).
281 // This may run arbitrary Python code and switch
282 // threads or greenlets!
283 Py_DECREF(to_del);
284 if (PyErr_Occurred()) {
285 PyErr_WriteUnraisable(nullptr);
286 PyErr_Clear();
287 }
288 }
289 }

333 ~ThreadState()
334 {
335 if (!PyInterpreterState_Head()) {
336 // We shouldn't get here (our callers protect us)
337 // but if we do, all we can do is bail early.
338 return;
339 }
340
341 // We should not have an "origin" greenlet; that only exists
342 // for the temporary time during a switch, which should not
343 // be in progress as the thread dies.
344 //assert(!this->switching_state.origin);
345
346 this->tracefunc.CLEAR();
347
348 // Forcibly GC as much as we can.
349 this->clear_deleteme_list(true);
350
351 // The pending call did this.
352 assert(this->main_greenlet->thread_state() == nullptr);
353
354 // If the main greenlet is the current greenlet,
355 // then we "fell off the end" and the thread died.
356 // It's possible that there is some other greenlet that
357 // switched to us, leaving a reference to the main greenlet
358 // on the stack, somewhere uncollectible. Try to detect that.
359 if (this->current_greenlet == this->main_greenlet && this->current_greenlet) {
360 assert(this->current_greenlet->is_currently_running_in_some_thread());
361 // Drop one reference we hold.
362 this->current_greenlet.CLEAR();
363 assert(!this->current_greenlet);
364 // Only our reference to the main greenlet should be left,
365 // But hold onto the pointer in case we need to do extra cleanup.
366 PyGreenlet* old_main_greenlet = this->main_greenlet.borrow();
367 Py_ssize_t cnt = this->main_greenlet.REFCNT();
368 this->main_greenlet.CLEAR();
369 if (ThreadState::_clocks_used_doing_gc != std::clock_t(-1)
370 && cnt == 2 && Py_REFCNT(old_main_greenlet) == 1) {
371 // Highly likely that the reference is somewhere on
372 // the stack, not reachable by GC. Verify.
373 // XXX: This is O(n) in the total number of objects.
374 // TODO: Add a way to disable this at runtime, and
375 // another way to report on it.
376 std::clock_t begin = std::clock();
377 NewReference gc(PyImport_ImportModule("gc"));
378 if (gc) {
379 OwnedObject get_referrers = gc.PyRequireAttr(ThreadState::get_referrers_name);
380 OwnedList refs(get_referrers.PyCall(old_main_greenlet));
381 if (refs && refs.empty()) {
382 assert(refs.REFCNT() == 1);
383 // We found nothing! So we left a dangling
384 // reference: Probably the last thing some
385 // other greenlet did was call
386 // 'getcurrent().parent.switch()' to switch
387 // back to us. Clean it up. This will be the
388 // case on CPython 3.7 and newer, as they use
389 // an internal calling conversion that avoids
390 // creating method objects and storing them on
391 // the stack.
392 Py_DECREF(old_main_greenlet);
393 }
394 else if (refs
395 && refs.size() == 1
396 && PyCFunction_Check(refs.at(0))
397 && Py_REFCNT(refs.at(0)) == 2) {
398 assert(refs.REFCNT() == 1);
399 // Ok, we found a C method that refers to the
400 // main greenlet, and its only referenced
401 // twice, once in the list we just created,
402 // once from...somewhere else. If we can't
403 // find where else, then this is a leak.
404 // This happens in older versions of CPython
405 // that create a bound method object somewhere
406 // on the stack that we'll never get back to.
407 if (PyCFunction_GetFunction(refs.at(0).borrow()) == (PyCFunction)green_switch) {
408 BorrowedObject function_w = refs.at(0);
409 refs.clear(); // destroy the reference
410 // from the list.
411 // back to one reference. Can *it* be
412 // found?
413 assert(function_w.REFCNT() == 1);
414 refs = get_referrers.PyCall(function_w);
415 if (refs && refs.empty()) {
416 // Nope, it can't be found so it won't
417 // ever be GC'd. Drop it.
418 Py_CLEAR(function_w);
419 }
420 }
421 }
422 std::clock_t end = std::clock();
423 ThreadState::_clocks_used_doing_gc += (end - begin);
424 }
425 }
426 }
427
428 // We need to make sure this greenlet appears to be dead,
429 // because otherwise deallocing it would fail to raise an
430 // exception in it (the thread is dead) and put it back in our
431 // deleteme list.
432 if (this->current_greenlet) {
433 this->current_greenlet->murder_in_place();
434 this->current_greenlet.CLEAR();
435 }
436
437 if (this->main_greenlet) {
438 // Couldn't have been the main greenlet that was running
439 // when the thread exited (because we already cleared this
440 // pointer if it was). This shouldn't be possible?
441
442 // If the main greenlet was current when the thread died (it
443 // should be, right?) then we cleared its self pointer above
444 // when we cleared the current greenlet's main greenlet pointer.
445 // assert(this->main_greenlet->main_greenlet == this->main_greenlet
446 // || !this->main_greenlet->main_greenlet);
447 // // self reference, probably gone
448 // this->main_greenlet->main_greenlet.CLEAR();
449
450 // This will actually go away when the ivar is destructed.
451 this->main_greenlet.CLEAR();
452 }
453
454 if (PyErr_Occurred()) {
455 PyErr_WriteUnraisable(NULL);
456 PyErr_Clear();
457 }
458
459 }

492 inline ThreadState& state()
493 {
494 // The main greenlet will own this pointer when it is created,
495 // which will be right after this. The plan is to give every
496 // greenlet a pointer to the main greenlet for the thread it
497 // runs in; if we are doing something cross-thread, we need to
498 // access the pointer from the main greenlet. Deleting the
499 // thread, and hence the thread-local storage, will delete the
500 // state pointer in the main greenlet.
501 if (this->_state == (ThreadState*)1) {
502 // XXX: Assuming allocation never fails
503 this->_state = new ThreadState;
504 // For non-standard threading, we need to store an object
505 // in the Python thread state dictionary so that it can be
506 // DECREF'd when the thread ends (ideally; the dict could
507 // last longer) and clean this object up.
508 }
509 if (!this->_state) {
510 throw std::runtime_error("Accessing state after destruction.");
511 }
512 return *this->_state;
513 }

516 {
517 return this->state();
518 }

521 {
522 return &this->state();
523 }

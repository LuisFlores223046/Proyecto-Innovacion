17// Define a Python object that goes in the Python thread state dict
18// when the greenlet thread state is created, and which owns the
19// reference to the greenlet thread local state.
20// When the thread state dict is cleaned up, so too is the thread
21// state. This works best if we make sure there are no circular
22// references to the thread state.
23typedef struct _PyGreenletCleanup {
24 PyObject_HEAD
25 ThreadStateCreator* thread_state_creator;
26} PyGreenletCleanup;
27
28static void
29cleanup_do_dealloc(PyGreenletCleanup* self)
30{
31 ThreadStateCreator* tmp = self->thread_state_creator;
32 self->thread_state_creator = nullptr;
33 if (tmp) {
34 delete tmp;
35 }
36}
37
38static void
39cleanup_dealloc(PyGreenletCleanup* self)
40{
41 PyObject_GC_UnTrack(self);
42 cleanup_do_dealloc(self);
43}
44
45static int
46cleanup_clear(PyGreenletCleanup* self)
47{
48 // This method is never called by our test cases.
49 cleanup_do_dealloc(self);
50 return 0;
51}
52
53static int
54cleanup_traverse(PyGreenletCleanup* self, visitproc visit, void* arg)
55{
56 if (self->thread_state_creator) {
57 return self->thread_state_creator->tp_traverse(visit, arg);
58 }
59 return 0;
60}

62static int
63cleanup_is_gc(PyGreenlet* UNUSED(self))
64{
65 return 1;

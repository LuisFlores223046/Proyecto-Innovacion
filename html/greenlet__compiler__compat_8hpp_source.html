60// CAUTION: MSVC is stupidly picky:
61//
62// "The compiler ignores, without warning, any __declspec keywords
63// placed after * or & and in front of the variable identifier in a
64// declaration."
65// (https://docs.microsoft.com/en-us/cpp/cpp/declspec?view=msvc-160)
66//
67// So pointer return types must be handled differently (because of the
68// trailing *), or you get inscrutable compiler warnings like "error
69// C2059: syntax error: ''"
70//
71// In C++ 11, there is a standard syntax for attributes, and
72// GCC defines an attribute to use with this: [[gnu:noinline]].
73// In the future, this is expected to become standard.
74
75#if defined(__GNUC__) || defined(__clang__)
76/* We used to check for GCC 4+ or 3.4+, but those compilers are
77 laughably out of date. Just assume they support it. */
78# define GREENLET_NOINLINE(name) __attribute__((noinline)) name
79# define GREENLET_NOINLINE_P(rtype, name) rtype __attribute__((noinline)) name
80# define UNUSED(x) UNUSED_ ## x __attribute__((__unused__))
81#elif defined(_MSC_VER)
82/* We used to check for && (_MSC_VER >= 1300) but that's also out of date. */
83# define GREENLET_NOINLINE(name) __declspec(noinline) name
84# define GREENLET_NOINLINE_P(rtype, name) __declspec(noinline) rtype name
